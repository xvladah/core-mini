# 1. Vytvoření složky logs a stop.txt, pokud neexistují
mkdir C:\MyService\logs
if (-not (Test-Path C:\MyService\stop.txt)) { New-Item C:\MyService\stop.txt -ItemType File }

# 2. Instalace služby
C:\nssm\nssm.exe install MyPHPService C:\php\php.exe C:\MyService\run.php

# 3. Nastavení pracovního adresáře
C:\nssm\nssm.exe set MyPHPService AppDirectory C:\MyService

# 4. Logování stdout a stderr s rotací
C:\nssm\nssm.exe set MyPHPService AppStdout C:\MyService\logs\service.log
C:\nssm\nssm.exe set MyPHPService AppStderr C:\MyService\logs\service_err.log
C:\nssm\nssm.exe set MyPHPService AppRotateFiles 1
C:\nssm\nssm.exe set MyPHPService AppRotateOnline 1
C:\nssm\nssm.exe set MyPHPService AppRotateBytes 10485760
C:\nssm\nssm.exe set MyPHPService AppRotateDelay 60

# 5. Restart při pádu služby
C:\nssm\nssm.exe set MyPHPService AppExit Default Restart
C:\nssm\nssm.exe set MyPHPService AppRestartDelay 5000
C:\nssm\nssm.exe set MyPHPService AppThrottle 2000

# 6. Spuštění služby
net start MyPHPService

# Povolení portu 9000 pro TCP příchozí připojení
New-NetFirewallRule -DisplayName "MyService TCP 9000" -Direction Inbound -LocalPort 9000 -Protocol TCP -Action Allow

Windows – NSSM
Pro každou službu:
Path: C:\php\php.exe
Arguments: C:\MyService\run.php
Environment → SERVICE_NAME → např. serviceA
I/O → Output file: C:\MyService\logs\serviceA.log
I/O → Error file: C:\MyService\logs\serviceA.err.log
Další služba → SERVICE_NAME=serviceB + vlastní logy.
